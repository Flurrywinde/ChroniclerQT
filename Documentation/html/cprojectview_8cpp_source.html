358

360  {
361  processed.prepend(bubble);
362 
363  QString indent_str = " ";
364  QString indent;
365 
366  // generate label...
367  if(LabelNeeded(bubble, bubbles))
368  cs += "\n\n*label " + MakeLabel(bubble, bubbles) + "\n";
369  else
370  indent = QString(indent_str).repeated(indent_level);
371 
372  // ------------ Start bubble ------------
373  if(bubble->getType() == Chronicler::Start)
374  {
375  CStartBubble *start = dynamic_cast<CStartBubble *>(bubble);
376  if(start->link())
377  cs += BubbleToChoiceScript(bubbles, processed, indent_level, start->link()->to());
378  else
379  cs += indent + "*finish\n";
380  }
381 
382  // ------------ Story bubble ------------
383  if(bubble->getType() == Chronicler::Story)
384  {
385  CStoryBubble *story = dynamic_cast<CStoryBubble *>(bubble);
386 
387  cs += indent + story->getStory().replace("\n", "\n" + indent) + "\n";
388 
389  if(story->link())
390  {
391  if(LabelNeeded(story->link()->to(), bubbles))
392  cs += indent + "*goto " + MakeLabel(story->link()->to(), bubbles);
393  // if(story->link()->to()->getType() == Chronicler::Story)
394  // cs += indent + "*page_break";
395  else
396  cs += "\n" + BubbleToChoiceScript(bubbles, processed, indent_level, story->link()->to());
397  }
398  else
399  cs += indent + "*finish";
400  }
401 
402  // ------------ Choice bubble ------------
403  else if(bubble->getType() == Chronicler::Choice)
404  {
405  CChoiceBubble *choice_bubble = dynamic_cast<CChoiceBubble *>(bubble);
406 
407  cs += indent + "*choice";
408 
409  // increase indent level
410  indent = QString(indent_str).repeated(++indent_level);
411 
412  for(CChoice *choice : choice_bubble->choiceBubbles())
413  {
414  // TODO add section for conditional choices
415  cs += "\n" + indent + "#" + choice->choice() + "\n";
416

418  cs += indent + indent_str + "*goto " + MakeLabel(choice->link()->to(), bubbles);
419  else if(choice->link())
420  cs += "\n" + BubbleToChoiceScript(bubbles, processed, indent_level + 1, choice->link()->to());
421  else
422  cs += indent + indent_str + "*finish";
423 
424  cs += "\n";
425  }
426  }
427 
428  // ------------ Action bubble ------------
429  else if(bubble->getType() == Chronicler::Action)
430  {
431  CActionBubble *action = dynamic_cast<CActionBubble *>(bubble);
432 
433  cs += indent + action->actionString().replace("\n", "\n" + indent) + "\n";

439  else

441  }
442 
443  // ------------ Condition bubble ------------
444  else if(bubble->getType() == Chronicler::Condition)
445  {
446  CConditionBubble *cb = dynamic_cast<CConditionBubble *>(bubble);
447 
448  cs += indent + "*if(" + cb->getCondition() + ")\n";
449 
450  // true
451  if(cb->trueLink())
452  {
453  if(LabelNeeded(cb->trueLink()->to(), bubbles))
454  cs += indent + indent_str + "*goto " + MakeLabel(cb->trueLink()->to(), bubbles);
455  else
456  cs += BubbleToChoiceScript(bubbles, processed, indent_level + 1, cb->trueLink()->to());
457  }
458  else

460

462
